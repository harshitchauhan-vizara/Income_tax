function drawBG(){
  BGX.clearRect(0,0,BGC.width,BGC.height);
  for(const p of BGPTS){p.x+=p.vx;p.y+=p.vy;
    if(p.x<0)p.x=BGC.width;if(p.x>BGC.width)p.x=0;
    if(p.y<0)p.y=BGC.height;if(p.y>BGC.height)p.y=0;
    const f=Math.sin(T*.003+p.ph)*.1+.9;
    BGX.beginPath();BGX.arc(p.x,p.y,p.r,0,6.28);
    BGX.fillStyle=`hsla(${p.h},70%,65%,${p.op*f})`;BGX.fill();}
  for(let i=0;i<BGPTS.length;i++)for(let j=i+1;j<BGPTS.length;j++){
    const dx=BGPTS[i].x-BGPTS[j].x,dy=BGPTS[i].y-BGPTS[j].y,d=Math.sqrt(dx*dx+dy*dy);
    if(d<85){BGX.beginPath();BGX.moveTo(BGPTS[i].x,BGPTS[i].y);BGX.lineTo(BGPTS[j].x,BGPTS[j].y);
      BGX.strokeStyle=`rgba(74,144,226,${.025*(1-d/85)})`;BGX.lineWidth=.3;BGX.stroke();}}
}

function r(a,b){return a+Math.random()*(b-a)}
function lp(a,b,t){return a+(b-a)*t}
function cl(v,lo,hi){return Math.max(lo,Math.min(hi,v))}

function dEye(ex,ey,blink,sq,wide,isL){
  G.save();G.translate(ex,ey);
  const eH=Math.max(.4,(1-blink)*(15+wide*3-sq*4));
  const eW=19+wide*1.5-sq;

  // Eye white
  G.save();
  G.beginPath();G.ellipse(0,0,eW,eH,0,0,6.28);G.clip();
  const ew=G.createRadialGradient(-1,-2,0,0,0,eW);
  ew.addColorStop(0,'#ffffff');ew.addColorStop(.7,'#f8f8ff');ew.addColorStop(1,'#eeeef8');
  G.fillStyle=ew;G.beginPath();G.ellipse(0,0,eW,eH,0,0,6.28);G.fill();

  if(eH>2){
    const iR=cl(12+wide*1.5-sq*2,3,13);
    const mGX=eW-iR-1,mGY=eH-iR-.5;
    const ix=cl(M.gazeX*3.5,-mGX,mGX),iy=cl(M.gazeY*2,-mGY,mGY);
    G.save();G.beginPath();G.arc(ix,iy,iR,0,6.28);G.clip();
    // Beautiful hazel iris with depth
    const ig=G.createRadialGradient(ix,iy-1,0,ix,iy,iR);
    ig.addColorStop(0,'#5a4a38');
    ig.addColorStop(.3,'#3a2a20');
    ig.addColorStop(.6,'#1e1410');
    ig.addColorStop(1,'#0c0806');
    G.fillStyle=ig;G.beginPath();G.arc(ix,iy,iR,0,6.28);G.fill();
    // Dark pupil
    G.beginPath();G.arc(ix,iy,iR*.55,0,6.28);G.fillStyle='#040302';G.fill();
    // ★ Large catchlight — glossy anime style
    G.save();G.translate(ix-iR*.28,iy-iR*.3);G.rotate(-.3);
    G.beginPath();G.ellipse(0,0,iR*.4,iR*.28,0,0,6.28);
    G.fillStyle='rgba(255,255,255,.98)';G.fill();G.restore();
    // Small secondary catchlight
    G.beginPath();G.arc(ix+iR*.42,iy+iR*.28,iR*.15,0,6.28);
    G.fillStyle='rgba(255,255,255,.65)';G.fill();
    G.restore();
  }
  G.restore();

  // Upper lid with lashes
  const lidY=-eH+(blink+sq*.3)*(eH*2+2);
  G.beginPath();
  G.moveTo(-eW-3,lidY-1);
  G.bezierCurveTo(-eW*.4,lidY-7,eW*.25,lidY-9,eW+4,lidY-2);
  G.lineTo(eW+4,lidY+3);
  G.bezierCurveTo(eW*.25,lidY-2,-eW*.4,lidY,-eW-3,lidY+3.5);
  G.closePath();G.fillStyle='#1a0c08';G.fill();

  // Lower lash line (soft)
  G.beginPath();G.moveTo(-eW+3,eH*.8);
  G.bezierCurveTo(-eW*.3,eH+2,eW*.3,eH+2,eW-3,eH*.8);
  G.strokeStyle='rgba(60,30,15,.35)';G.lineWidth=.9;G.stroke();

  G.restore();
}

function dLash(ex,ey,blink,sq,wide){
  if(blink>.85)return;
  G.save();G.translate(ex,ey);
  const eH=Math.max(.5,(1-blink)*(15+wide*3-sq*4));
  const eW=19+wide*1.5-sq;
  const lidY=-eH+(blink+sq*.3)*(eH*2+2);
  const al=cl(1-blink*1.5,0,1);
  G.strokeStyle=`rgba(10,5,2,${al})`;
  const ld=[[-eW-1,-1.5,-9,-1.2],[-eW*.6,-2.5,-8.5,-.9],[-eW*.28,-3.2,-9,-.3],
    [0,-3.5,-9.5,0],[eW*.28,-3.2,-9,.3],[eW*.6,-2.5,-8.5,.9],[eW+1,-1.5,-8.5,1.2]];
  ld.forEach(([lx,ly,ly2,curl])=>{
    G.beginPath();G.moveTo(lx,lidY+ly);
    G.bezierCurveTo(lx+curl*1,lidY+ly2,lx+curl*1.8,lidY+ly2-2,lx+curl*2.4,lidY+ly2-3.5);
    G.lineWidth=1.3+(1-Math.abs(lx)/eW)*.6;G.stroke();
  });
  G.restore();
}

let aT=null,tI=null,cSt='idle';
const MICO='<rect x="9" y="2" width="6" height="11" rx="3"/><path d="M19 10v2a7 7 0 01-14 0v-2"/><line x1="12" y1="19" x2="12" y2="23"/><line x1="8" y1="23" x2="16" y2="23"/>';

function S(s){
  cSt=s;VSTATE=s;
  if(aT){clearTimeout(aT);aT=null;}if(tI){clearInterval(tI);tI=null;}
  if('speechSynthesis' in window)window.speechSynthesis.cancel();
  LS.active=false;
  ['idle','listening','thinking','speaking'].forEach(n=>{
    const b=document.getElementById('b-'+n);if(b)b.className=n===s?'on':'';
  });
  document.querySelector('.panel').className='panel state-'+s;

  const stn=document.getElementById('stn'),wv=document.getElementById('wv');
  const pd=document.getElementById('pdot'),pl=document.getElementById('plbl');
  const sd2=document.getElementById('sdot2');
  const mc=document.getElementById('mc'),msv=document.getElementById('msv'),ml=document.getElementById('ml');
  const resp=document.getElementById('resp');
  function sd(el,c){el.className='pill-dot'+(c?' '+c:'');}

  if(s==='idle'){
    setEmotion('happy');
    stn.textContent='IDLE';stn.className='st-n';
    wv.className='wv';sd(pd,'');sd(sd2,'');
    pl.textContent='STANDBY';pl.className='pill-lbl';
    mc.className='mc';msv.innerHTML=MICO;msv.setAttribute('stroke','rgba(74,144,226,.85)');
    ml.textContent='PRESS TO SPEAK';ml.style.color='';
    resp.classList.remove('on');LS.active=false;

  } else if(s==='listening'){
    setEmotion('curious');
    stn.textContent='LISTENING';stn.className='st-n c-blue';
    wv.className='wv on';wv.querySelectorAll('.wb').forEach(b=>b.style.background='var(--blue2)');
    sd(pd,'a-blue');sd(sd2,'a-blue');
    pl.textContent='RECEIVING';pl.className='pill-lbl a-blue';
    mc.className='mc on';msv.innerHTML=MICU;msv.setAttribute('stroke','#fff');
    ml.textContent='LISTENING...';ml.style.color='var(--blue2)';
    resp.classList.remove('on');LS.active=false;

  } else if(s==='thinking'){
    setEmotion('thinking');
    stn.textContent='THINKING';stn.className='st-n c-cyan';
    wv.className='wv';sd(pd,'a-cyan');sd(sd2,'a-cyan');
    pl.textContent='PROCESSING';pl.className='pill-lbl a-cyan';
    mc.className='mc';msv.innerHTML=MICO;msv.setAttribute('stroke','rgba(74,144,226,.85)');
    ml.textContent='THINKING...';ml.style.color='var(--cyan)';
    resp.classList.remove('on');LS.active=false;
    aT=setTimeout(()=>S('speaking'),3200);

  } else if(s==='speaking'){
    setEmotion('speaking');
    stn.textContent='SPEAKING';stn.className='st-n c-gold';
    wv.className='wv on';wv.querySelectorAll('.wb').forEach(b=>b.style.background='var(--gold)');
    sd(pd,'a-gold');sd(sd2,'a-gold');
    pl.textContent='RESPONDING';pl.className='pill-lbl a-gold';
    mc.className='mc';msv.innerHTML=MICO;msv.setAttribute('stroke','rgba(74,144,226,.85)');
    ml.textContent='PRESS TO STOP';ml.style.color='var(--gold)';
    const txt="Hi! I've carefully analyzed your request and I'm excited to share my findings! The results are fascinating — there are some really interesting patterns here that I think you'll love. Let me walk you through everything step by step!";
    resp.classList.add('on');

    // Drive lip sync from character index during speech
    LS.seq=txt.split('').map(c=>cp(c));
    LS.idx=0;LS.timer=0;LS.active=true;LS.fd=3;

    // Typewriter effect
    const rt=document.getElementById('rtxt');rt.innerHTML='';let i=0;
    tI=setInterval(()=>{
      if(i<txt.length){
        LS.idx=i; // sync lipsync to typewriter position
        rt.innerHTML=txt.slice(0,++i)+'<span class="cur"></span>';
      } else{clearInterval(tI);rt.innerHTML=txt;}
    },26);

    // ── Web Speech API TTS ──
    if('speechSynthesis' in window){
      window.speechSynthesis.cancel(); // stop any previous
      const utt=new SpeechSynthesisUtterance(txt);
      utt.rate=0.95;
      utt.pitch=1.15;
      utt.volume=1.0;

      // Pick a female voice if available
      function pickVoice(){
        const voices=window.speechSynthesis.getVoices();
        // Prefer: Google UK English Female, Samantha, Karen, Moira, or any female voice
        const preferred=['Google UK English Female','Samantha','Karen','Moira','Tessa','Zira'];
        for(const name of preferred){
          const v=voices.find(v=>v.name===name);
          if(v)return v;
        }
        // Fallback: any voice with 'female' in name or en-US/en-GB
        const female=voices.find(v=>/female/i.test(v.name));
        if(female)return female;
        const en=voices.find(v=>/en[-_](US|GB|AU)/i.test(v.lang));
        if(en)return en;
        return voices[0]||null;
      }

      function startSpeech(){
        const v=pickVoice();
        if(v)utt.voice=v;
        // Boundary event — advance lip sync to word position
        utt.onboundary=function(e){
          if(e.name==='word'){LS.idx=Math.min(e.charIndex,txt.length-1);}
        };
        utt.onend=function(){
          LS.active=false;
          if(cSt==='speaking')aT=setTimeout(()=>S('idle'),800);
        };
        utt.onerror=function(){
          LS.active=false;
        };
        window.speechSynthesis.speak(utt);
      }

      // Voices may not be loaded yet
      if(window.speechSynthesis.getVoices().length>0){
        startSpeech();
      } else {
        window.speechSynthesis.onvoiceschanged=function(){startSpeech();};
      }

      // Store utterance so we can cancel on state change
      window._hanaUtt=utt;
    } else {
      // No TTS support — fall back to timer
      aT=setTimeout(()=>S('idle'),11000);
    }
  }
}
